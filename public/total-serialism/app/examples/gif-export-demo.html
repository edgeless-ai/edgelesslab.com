Integration Example

// 1. Include the GIF exporter script
<script src="gif-exporter.js"></script>

// 2. Create exporter instance
const exporter = new GifExporter({
  fps: 30,
  duration: 3,
  quality: 10,
  width: 600,
  height: 600
});

// 3. Initialize with your canvas
exporter.init(canvas, {
  onProgress: (data) => {
    console.log(`Progress: ${Math.round(data.progress * 100)}%`);
  },
  onComplete: (result) => {
    console.log('GIF created!', result);
  }
});

// 4. Start/stop recording
exporter.start(); // Begin recording
exporter.stop();  // End and process

// Or use the quick setup for p5.js:
const exporter = setupGifExporter(sketch, {
  fps: 30,
  duration: 3,
  autoDownload: true,
  recordKey: 'r'  // Press 'r' to toggle recording
});
